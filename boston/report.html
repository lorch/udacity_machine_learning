def performance_metric2(y_true, y_predict):
    """"""
    
    mean = np.mean(y_true)
    
    sse = 0.0
    sst = 0.0
    for i in range(0, len(y_true)):
        sst += (y_true[i] - mean) ** 2
        sse += (y_true[i] - y_predict[i]) ** 2
        
    score = 1 - float(sse)/float(sst)

    return score

def cross_val_score(estimator, X, y, scoring = performance_metric, cv=3):
    """  """
    k1 = len(X)/3
    k2 = 2 * k1

    X1= X[:k1]
    X2 = X[k1:k2]
    X3 = X[k2:]

    y1 = y[:k1]
    y2 = y[k1:k2]
    y3 = y[k2:]
    
    estimator.fit(np.append(X2,X3, axis =0),np.append(y2,y3,axis =0))
    pred1 = estimator.predict(X1)
    
    estimator.fit(np.append(X1,X3, axis =0),np.append(y1,y3,axis =0))
    pred2 = estimator.predict(X2)
    
    estimator.fit(np.append(X1,X2, axis =0),np.append(y1,y2,axis =0))
    pred3 = estimator.predict(X3)
    
    scores = [performance_metric(y1,pred1), performance_metric(y2,pred2), performance_metric(y3,pred3)]
    
    return scores

def fit_model2(X, y):
    """  [X,y]"""
    
    groups = {}
    
    #
    for depth in range(1,11):
        reg = DecisionTreeRegressor(max_depth = depth)
        score = cross_val_score(reg, X, y, scoring = performance_metric, cv =3)
        groups[depth] = np.mean(score)
    
    maxScoreGroup = max(groups.items(), key=lambda x: x[1])
    best_estimator = DecisionTreeRegressor(max_depth = maxScoreGroup[0])
    
    return best_estimator
